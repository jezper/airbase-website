# Phase 3: Feed (Home Page) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the placeholder feed on the home page with real card components (Release, Show, Note, Article) displaying actual data, unified into a reverse-chronological stream with client-side filter pills.

**Architecture:** Server component loads all content (releases, shows, sample posts) and passes them as a unified sorted array to a client component that handles filtering. Each post type has its own card component. Filter pills toggle which types are visible. The hero and stats bar from Phase 1 remain at the top.

**Tech Stack:** Next.js App Router (server + client components), TypeScript, Tailwind CSS v4, Lucide React icons

**Design reference:** `docs/superpowers/specs/2026-04-12-visual-direction-design.md` — Feed section

**Existing code:**
- Types: `src/types/content.ts` (Release, Show, PressFeature, PostType, FeedPost)
- Loaders: `src/lib/releases.ts`, `src/lib/shows.ts`
- Design tokens: `src/app/globals.css` (all Tailwind utilities available)
- Current page: `src/app/page.tsx` (hero + stats + placeholder cards)

Create `src/lib/feed.ts`:

```typescript
import { getAllReleases } from "./releases";
import { getAllShows } from "./shows";
import type { FeedItem, FeedRelease, FeedShow, FeedNote, FeedArticle } from "@/types/content";

// Sample notes for Phase 3 demo (will be replaced by MDX posts in Phase 7)
const SAMPLE_NOTES: FeedNote[] = [
  {
    type: "note",
    date: "2026-04-10",
    data: {
      body: "Just finished a new sketch in the studio. Something different this time. Pushing the tempo, pushing the melody. Emotion first, details later.",
    },
  },
  {
    type: "note",
    date: "2026-03-15",
    data: {
      body: "Listening back to old Ozone tracks. The naivety in those productions is what made them work. You can't fake that energy.",
    },
  },
];

const SAMPLE_ARTICLES: FeedArticle[] = [
  {
    type: "article",
    date: "2026-02-10",
    data: {
      title: "People didn't fall in love with perfect production",
      excerpt: "An in-depth conversation with Beatportal about the return, the hiatus, and why melody still wins.",
      slug: "beatportal-interview-2026",
    },
  },
];

export async function getFeedItems(limit?: number): Promise<FeedItem[]> {
  const [releases, shows] = await Promise.all([
    getAllReleases(),
    getAllShows(),
  ]);

  const feedReleases: FeedRelease[] = releases.map((r) => ({
    type: "release",
    date: r.date,
    data: r,
  }));

  const feedShows: FeedShow[] = shows
    .filter((s) => s.date !== null)
    .map((s) => ({
      type: "show",
      date: s.date!,
      data: s,
    }));

  const allItems: FeedItem[] = [
    ...feedReleases,
    ...feedShows,
    ...SAMPLE_NOTES,
    ...SAMPLE_ARTICLES,
  ];

  allItems.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());

  if (limit) {
    return allItems.slice(0, limit);
  }

  return allItems;
}

export function getFilterTypes(): { value: FeedItem["type"] | "all"; label: string }[] {
  return [
    { value: "all", label: "All" },
    { value: "release", label: "Releases" },
    { value: "show", label: "Shows" },
    { value: "note", label: "Notes" },
    { value: "article", label: "Articles" },
  ];
}
```

```typescript
import type { Show } from "@/types/content";

function formatShowDate(show: Show): { day: string; month: string; year: string } {
  if (show.date) {
    const d = new Date(show.date);
    return {
      day: d.getDate().toString().padStart(2, "0"),
      month: d.toLocaleString("en", { month: "short" }).toUpperCase(),
      year: d.getFullYear().toString(),
    };
  }
  return {
    day: "—",
    month: show.year_approx?.slice(0, 4).toUpperCase() ?? "TBA",
    year: "",
```

```typescript
    <div className="flex gap-1.5 overflow-x-auto scrollbar-none">
```

This receives all feed items from the server component, manages filter state, and renders the appropriate card for each item.

Create `src/components/feed/feed.tsx`:

```typescript
"use client";

import { useState } from "react";
import type { FeedItem } from "@/types/content";
import { FilterPills } from "./filter-pills";
import { ReleaseCard } from "./release-card";
import { ShowCard } from "./show-card";
import { NoteCard } from "./note-card";
import { ArticleCard } from "./article-card";

const FILTERS: { value: FeedItem["type"] | "all"; label: string }[] = [
  { value: "all", label: "All" },
  { value: "release", label: "Releases" },
  { value: "show", label: "Shows" },
  { value: "note", label: "Notes" },
  { value: "article", label: "Articles" },
];

function FeedCard({ item }: { item: FeedItem }) {
  switch (item.type) {
    case "release":
      return <ReleaseCard release={item.data} />;
    case "show":
      return <ShowCard show={item.data} />;
    case "note":
      return <NoteCard note={item} />;
    case "article":
      return <ArticleCard article={item} />;
    default:
      return null;
  }
}

export function Feed({ items }: { items: FeedItem[] }) {
  const [filter, setFilter] = useState<FeedItem["type"] | "all">("all");

  const filtered = filter === "all"
    ? items
    : items.filter((item) => item.type === filter);

  return (
    <section className="px-6 md:px-12 py-12" aria-label="Feed">
      <div className="flex flex-col sm:flex-row items-start sm:items-center justify-between gap-4 mb-8">
        <h2 className="font-body text-[13px] font-bold uppercase tracking-[0.12em] text-text-faint">
          Feed
        </h2>
        <FilterPills filters={FILTERS} active={filter} onChange={setFilter} />
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {filtered.map((item, i) => (
          <div
            key={`${item.type}-${item.date}-${i}`}
            className={item.type === "release" ? "md:row-span-2" : ""}
          >
            <FeedCard item={item} />
          </div>
        ))}
      </div>

      {filtered.length === 0 && (
        <p className="text-center text-text-faint font-body py-16">
          No posts to show right now.
        </p>
      )}
    </section>
  );
}
```
